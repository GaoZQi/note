        }
        if (minute == 60)

            minute = 0;
            hour++;
        }
        if (hour == 24)
        {
            hour = 0;
            day++;
        }
        if (day == month_day[month])
        {
            day = 0;
            month++;
        }
        if (month > 12)
        {
            month = 1;
            year++;
        }
    }
    if (flag_beep == 1) // Á≠â‰∫é1Êó∂ËúÇÈ∏£Âô®300msÂìç‰∏Ä‰∏ã
    {
        if (t1 == 15) // 20ms*15=300ms

            t1 = 0;
            beep = ~beep;
        }
    }
    else
    {
        beep = 1;
    }
}
